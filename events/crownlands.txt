# Robert Baratheon Dies
country_event = {
	id = crownlands.1
	title = crownlands.1.t
	desc = crownlands.1.d
	picture = GFX_report_event_europe_funeral

	is_triggered_only = yes

	fire_only_once = yes

	mean_time_to_happen = {
		days = 20
	}

	immediate = {
		set_country_flag = CRO_king_robert_died
		unlock_national_focus = CRO_robert_died
	}

	option = {
		name = crownlands.1.a
		add_stability = -0.1
		kill_country_leader = yes
		add_country_leader_role = {
			character = CRO_joffrey_baratheon
			country_leader = {
				expire = "1965.1.1"
				ideology = despotism
				traits = { incompetent_leader }
			}
			promote_leader = yes
		}
		add_field_marshal_role = {
			character = CRO_joffrey_baratheon
			traits = { trait_reckless }
			skill = 1
			attack_skill = 3
			defense_skill = 1
			planning_skill = 1
			logistics_skill = 1
		}
		add_advisor_role = {
			character = CRO_cersei_lannister
			activate = yes
			advisor = {
				slot = political_advisor
				idea_token = CRO_cersei_lannister
				can_be_fired = no
				traits = {
					queen_regent
				}
			}
		}
		WES = {
			add_opinion_modifier = {
				modifier = accepted_regency
				target = ROOT
			}
		}
		NOR = {
			add_opinion_modifier = {
				modifier = declined_regency
				target = ROOT
			}
		}
		hidden_effect = {
			news_event = { id = news.504 days = 1 }
		}
		country_event = { id = crownlands.2 days = 1 }	#Deal with the Starks
	}
	option = {	#TODO - for a latter date
		name = crownlands.1.b
		add_stability = -0.1
		kill_country_leader = yes
		add_country_leader_role = {
			character = NOR_ned_stark
			country_leader = {
				expire = "1965.1.1"
				ideology = despotism
				traits = { incompetent_leader }
			}
			promote_leader = yes
		}
		add_field_marshal_role = {
			character = NOR_ned_stark
			traits = { trait_reckless }
			skill = 1
			attack_skill = 3
			defense_skill = 1
			planning_skill = 1
			logistics_skill = 1
		}
		NOR = {
			add_opinion_modifier = {
				modifier = accepted_regency
				target = ROOT
			}
		}
		WES = {
			add_opinion_modifier = {
				modifier = declined_regency
				target = ROOT
			}
		}
		hidden_effect = {
			news_event = { id = news.504 days = 1 }
		}
		country_event = { id = crownlands.3 days = 1 }	#Deal with the Lannister
	}
}

# New Hand Appointed
country_event = {
	id = crownlands.5
	title = crownlands.5.t
	desc = crownlands.5.d
	picture = GFX_report_event_NOR_royal_family_exiled

	is_triggered_only = yes

	option = {
		name = crownlands.5.a
		if = {
			limit = { tag = NOR }
			NOR_ned_stark = {
				add_trait = {
					ideology = despotism
					trait = hand_of_the_king
				}
			}
		}
		else_if = {
			limit = { tag = WES }
			WES_tywin_lannister = {
				add_trait = {
					ideology = despotism
					trait = hand_of_the_king
				}
			}
		}
		else_if = {
			limit = { tag = STO }
			STO_renly_baratheon = {
				add_trait = {
					ideology = despotism
					trait = hand_of_the_king
				}
			}
		}
		else = {
			custom_effect_tooltip = report_bug_tt
		}
		hidden_effect = {
				news_event = { id = news.511 days = 7 }
		}
	}
}

# Random Hunting Event
country_event = {
	id = crownlands.7
	title = crownlands.7.t
	desc = crownlands.7.d
	picture = GFX_report_event_royal_visit

	trigger = {
		tag = CRO
		date > 1936.1.7
		has_country_leader = {
			character = CRO_robert_baratheon
			ruling_only = yes
		}
		NOT = { has_country_flag = CRO_first_fire_hunting }
	}

	immediate = {
		hidden_effect = {
			add_timed_idea = { idea = CRO_king_hunting_id days = 30 }
			activate_mission = CRO_king_hunting
		}
	}

	option = {
		name = crownlands.7.a
		effect_tooltip = {
			activate_mission = CRO_king_hunting
			add_timed_idea = { idea = CRO_king_hunting_id days = 30 }
		}
		hidden_effect = {
			set_country_flag = CRO_first_fire_hunting
			country_event = { id = crownlands.6 days = 60 }
		}
		custom_effect_tooltip = CRO_king_hunting_tt
	}
}
