GOT_start_techs = {

set_rule = { can_create_factions = no }

	if = {
		limit = {
			OR = {
				tag = VAL
				tag = NOR
				tag = WES
				tag = DOR
				tag = RIV
				tag = STO
				tag = REA
				tag = CRO
				tag = BRA
				tag = VOL
				tag = YTI
				tag = ASS
				tag = QAR
			}
		}
		GOT_start_techs_A = yes
	}
	else_if = {
		limit = {
			OR = {
				tag = IRO
				tag = DRA
				tag = STE 
				tag = BWB
				tag = BOL
				tag = SUI
				tag = LOR
				tag = QOH
				tag = MYR
				tag = TYR
				tag = IBB
				tag = TOL
				tag = MAN
			}
		}
		GOT_start_techs_B = yes
	}
	else = {
		GOT_start_techs_C = yes
	}
	
	if = {
		limit = {
			tag = TAR
		}
		GOT_bonus_techs_TAR = yes
	}
	
	if = {
		limit = {
			tag = DOT
		}
		GOT_bonus_techs_DOT = yes
	}
	
	if = {
		limit = {
			tag = IRO
		}
		GOT_bonus_techs_IRO = yes
	}

}

		basic_train = 1
		tech_trucks = 1
		interwar_antitank = 1
		interwar_antiair = 1
	}
	if = {
		limit = {
			has_dlc = "No Step Back"
		}
		set_technology = {
			gwtank_chassis = 1
		}
	}
	if = {
		limit = {
			NOT = { has_dlc = "No Step Back" }
		}
		set_technology = {
			gwtank = 1
		}
	}
	if = {
		limit = {
			NOT = { tag = DOT }
		}
		if = {
			limit = {
				has_dlc = "Man the Guns"
			}
			set_technology = {
				early_ship_hull_light = 1
				early_ship_hull_cruiser = 1
				early_ship_hull_heavy = 1
				#early_ship_hull_carrier = 1 #Only Targs get them at game start
				early_ship_hull_submarine = 1
				mtg_transport = 1
				basic_battery = 1
				basic_torpedo = 1
			}
		}
		if = {
			limit = {
				NOT = { has_dlc = "Man the Guns" }
			}
			set_technology = {
				early_destroyer = 1
				early_light_cruiser = 1
				early_heavy_cruiser = 1
				early_battlecruiser = 1
				early_battleship = 1
				#early_carrier = 1 #Only Targs get them at Game Start
				early_submarine = 1
				transport = 1
			}
		}
		if = {
			limit = {
				has_dlc = "La Resistance"
			}
			set_technology = {
				armored_car1 = 1
			}
		}
	}
	
}
